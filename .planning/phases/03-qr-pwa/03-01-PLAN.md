1. QR URLは `/u/${urlId}/p/${shareToken}` 形式（現在表示ペルソナのshareTokenを使う）。window.location.origin はuseEffect後にのみ取得
2. SSR guard必須: `const [mounted, setMounted] = useState(false)` + `useEffect(() => { setMounted(true) }, [])` — マウント前は `<div className="w-48 h-48 bg-gray-100 rounded-lg animate-pulse" />` を返す
3. `import { Sheet } from 'react-modal-sheet'` (名前付きエクスポート、デフォルトエクスポートではない — Phase 0確認済みパターン)
4. QRCodeSVG props: `size={220}`, `level="M"`, `marginSize={4}`, `bgColor="#FFFFFF"`, `fgColor="#000000"`
5. ボトムシート構造: `detent="content"` で高さをコンテンツに合わせる
6. Props interface: `interface QRBottomSheetProps { isOpen: boolean; onClose: () => void; urlId: string; shareToken: string; displayName: string }`

コンポーネント構造:
```tsx
import { Sheet } from 'react-modal-sheet'
import { QRCodeSVG } from 'qrcode.react'
import { useState, useEffect } from 'react'

export function QRBottomSheet({ isOpen, onClose, urlId, shareToken, displayName }: QRBottomSheetProps) {
  const [mounted, setMounted] = useState(false)
  const [origin, setOrigin] = useState('')
  useEffect(() => {
    setMounted(true)
    setOrigin(window.location.origin)
  }, [])

  const qrUrl = origin ? `${origin}/u/${urlId}/p/${shareToken}` : ''

  return (
    <Sheet isOpen={isOpen} onClose={onClose} detent="content">
      <Sheet.Container>
        <Sheet.Header />
        <Sheet.Content>
          <div className="flex flex-col items-center gap-4 p-6 pb-10">
            <p className="text-sm text-gray-500">{displayName} のQRコード</p>
            {!mounted || !qrUrl ? (
              <div className="w-[220px] h-[220px] bg-gray-100 rounded-lg animate-pulse" />
            ) : (
              <QRCodeSVG value={qrUrl} size={220} level="M" marginSize={4} bgColor="#FFFFFF" fgColor="#000000" />
            )}
            <p className="text-xs text-gray-400 break-all text-center max-w-[240px]">{qrUrl}</p>
          </div>
        </Sheet.Content>
      </Sheet.Container>
      <Sheet.Backdrop onTap={onClose} />
    </Sheet>
  )
}
```
