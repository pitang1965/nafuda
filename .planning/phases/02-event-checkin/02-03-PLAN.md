const getOptionalSession = createServerFn({ method: 'GET' })
  .handler(async () => {
    const request = getRequest()
    // getSession は未認証の場合 null を返す（throw redirect しない）

**ルート定義:**
```typescript
export const Route = createFileRoute('/e/$slug')({
  loader: async ({ params }) => {
    const [data, session] = await Promise.all([
      getEventParticipants({ data: { slug: params.slug } }),
      getOptionalSession(),
    ])
    return { data, isLoggedIn: !!session?.user }
  },
  component: EventPage,
})
```

**コンポーネント実装:**
```typescript
function EventPage() {
  const { data, isLoggedIn } = Route.useLoaderData()

  if (!data) {
    return (
      <div className="...">
        <h1>イベントが見つかりません</h1>
        <p>URLを確認してください。</p>
      </div>
    )
  }

  return (
    <div className="...">
      {/* イベント情報ヘッダー */}
      <div>
        <h1>{data.event.name}</h1>
        <p>{data.event.venueName}</p>
        <p>{new Date(data.event.eventDate).toLocaleDateString('ja-JP')}</p>
      </div>

      {/* 参加者カウント */}
      <p>{data.participants.length}人がチェックイン中</p>

      {/* 未ログイン時の注意書き */}
      {!isLoggedIn && (
        <p className="...">
          <Link to="/login">ログイン</Link>するとプロフィールを閲覧できます
        </p>
      )}

      {/* 参加者グリッド */}
      <div className="grid grid-cols-2 sm:grid-cols-3 gap-3">
        {data.participants.map((p) => (
          <ParticipantCard
            key={p.checkinId}
            displayName={p.displayName}
            avatarUrl={p.avatarUrl}
            // OSHI-05: ログイン済みの場合のみ profileHref を渡す
            profileHref={isLoggedIn ? `/u/${p.urlId}/p/${p.shareToken}` : undefined}
          />
        ))}
      </div>

      {data.participants.length === 0 && (
        <p>現在チェックイン中の参加者はいません</p>
      )}
    </div>
  )
}
```

**必須インポート:**

- `auth` from `../../server/auth`
- `getEventParticipants` from `../../server/functions/event`
- `ParticipantCard` from `../../components/ParticipantCard`
- `Link` from `@tanstack/react-router`

**注意:**
