```typescript
export function generateSlug(eventName: string, eventDate: string): string {
  const datePart = eventDate.replace(/-/g, '')
  const namePart = eventName
    .toLowerCase()
    .replace(/[^\w\s-]/g, '')
    .replace(/[\s_]+/g, '-')
    .replace(/-+/g, '-')
    .slice(0, 50)
  return `${namePart}-${datePart}`
}
```
